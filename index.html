How to Use Maelyn API

Simple integration for your applications

JavaScript 
// Example: Using Maelyn API for AI chat responses
const axios = require('axios');

// Function to get AI response
async function getChatResponse(message) {
  try {
    const response = await axios.post('https://api.maelyn.tech/v1/ai/chat', {
      query: message,
      name: 'Assistant'
    }, {
      headers: {
        'mg-apikey': 'YOUR_API_KEY'
      }
    });
    
    return response.data.result;
  } catch (error) {
    console.error('Error:', error.message);
    return 'Sorry, I encountered an error.';
  }
}

// Example usage
async function main() {
  const userMessage = 'Hello, how are you?';
  const aiResponse = await getChatResponse(userMessage);
  console.log(aiResponse);
}
